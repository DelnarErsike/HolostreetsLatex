% Options.tex is executed before options are executed (duh) and before any packages are imported
% Use it for setting config options, but if you need to set more complex things post-packages, do it
% in definitions.tex
%
% Default options are set in shadowrun.cls

% This is a special option we use for when we build overleaf-extractable builds
% The install script will alter it for the overleaf specific folders

\DeclareOption{purple}{\def\@themeColour{purple}}
\DeclareOption{red}{\def\@themeColour{red}}
\DeclareOption{blue}{\def\@themeColour{blue}}

\DeclareOption{titlepage-seamless}
{
    \def\@titlepageseamless{}
    \ifdefined\@noimage
        \PackageError{shadowrun}{Cannot use both 'titlepage-seamless' and 'no-images' at the same time}
    \fi
}
\DeclareOption{titlepage-no-cgl}{\def\@titlepagenocgl{}}
\DeclareOption{no-images}{
    \def\@noimage{}
    \ifdefined\@titlepageseamless
        \PackageError{shadowrun}{Cannot use both 'titlepage-seamless' and 'no-images' at the same time}
    \fi
    \def\@titlepagenocgl{}
}
